Drawer menu

Quickly build a Drawer(Sliding) menu.

HTML

        
          <nav id="drawermenu" data-toggle="close">
            <div class="headermenu">
              ...
            </div>
            <div class="slidermenu">
              ...
            </div>
            <div class="overlay"></div>
          </nav>

JS

        
          $('#drawermenu').essence_drawermenu({
            'header_menu_color':'#FFF',
            'header_menu_bgcolor':'#1b3e57',
            'slider_menu_color':'#FFF',
            'slider_menu_bgcolor':'#183041'
          });
